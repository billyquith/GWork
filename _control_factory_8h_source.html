66  template <typename ENUM>
67  class ValueEnumMap
68  {
69  public:
70  typedef ENUM EnumType;
71  typedef unsigned int Count_t;
72  typedef ValueEnumMapItem<EnumType> Enum;
73  static const int Undefined = -1;
74 
75  ValueEnumMap(const Enum *pairs, Count_t count)
76  : m_enums(pairs)
77  , m_enumCount(count)
78  {}
79 
80  size_t GetEnumCount() const { return m_enumCount; }
81 
82  const char* GetNameByIndex(Count_t index) const
83  {
84  return m_enums[index].name;
85  }
86 
87  size_t GetIndexByName(const char *name) const
88  {
89  for (Count_t i=0; i < m_enumCount; ++i)
90  {
91  if (strcmp(m_enums[i].name, name)==0)
92  return i;
93  }
94  return Undefined;
95  }
96 
97  ENUM GetValueByName(const char *name, EnumType defaultValue = Undefined) const
98  {
99  const size_t i = GetIndexByName(name);
100  return i!=Undefined ? m_enums[i].value : defaultValue;
101  }
102 
103  const char* GetNameByValue(EnumType value, Count_t defaultIndex=0) const
104  {
105  for (size_t i=0; i < m_enumCount; ++i)
106  {
107  const Enum &item = m_enums[i];
108  if (item.value == value)
109  return item.name; // We found the item.
110  }
111  return m_enums[defaultIndex].name; // Not found. Return default indexed.
112  }
113 
114  private:
115  const Enum *m_enums;
116  const Count_t m_enumCount;
117  };
